# 선언, 확인, 제거하기

```bash title="" {} // showLineNumbers{number}

# 선언
firstVariable=hello

# 확인
echo $firstVariable # $ 기호를 붙인다.
set # 모든 변수 목록 확인

# 제거
unset $firstVariable
```
# 환경 변수

>[!hint] 환경 변수는 프로그램 전체에 영향을 주는 변수이다.

```bash title="" {} // showLineNumbers{number}
export VARIABLE=hello
echo $VARIABLE

env # 모든 환경 변수 조회

echo $PATH # 명령어 탐색 경로
echo $HOME # cd ~ 에 적용되는 홈 디렉토리
echo $SHELL
echo $USER
```
